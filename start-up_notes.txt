
# Start group of docker container wth docker compose:
$ docker-compose -f project-z-compose.yaml up -d

# interactive shell with container:
$ docker exec -ti <container name> /bin/bash

# PSQL
$ psql -h localhost -p 5432 -U postgres -W

# list volumes on host (persistent data)
$ sudo ls -l /var/lib/docker/volumes

# interactive mode. - on running container.
$ docker exec -ti <container name> /bin/bash ; (sh)
